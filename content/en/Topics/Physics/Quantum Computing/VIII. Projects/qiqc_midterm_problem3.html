theta = 1/3

qpe = QuantumCircuit(4, 3)
qpe.x(3)
for qubit in range(3):
    qpe.h(qubit)
repetitions = 1
for counting_qubit in range(3):
    for i in range(repetitions):
        qpe.cp(2*pi*theta, counting_qubit, 3); # This is C-U
    repetitions *= 2
qpe.draw(output='mpl')
